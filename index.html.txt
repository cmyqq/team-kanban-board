    <script>
        // 计算100-999的所有水仙花数
        function findNarcissisticNumbers() {
            const result = [];
            for (let i = 100; i <= 999; i++) {
                // 拆分百位、十位、个位
                const hundreds = Math.floor(i / 100);
                const tens = Math.floor((i % 100) / 10);
                const units = i % 10;
                // 计算各位数字立方和
                const sum = Math.pow(hundreds, 3) + Math.pow(tens, 3) + Math.pow(units, 3);
                // 判断是否为水仙花数
                if (sum === i) {
                    result.push(i);
                }
            }
            return result;
        }

        // 控制台打印结果（验证用）
        console.log('100-999的水仙花数：', findNarcissisticNumbers());
